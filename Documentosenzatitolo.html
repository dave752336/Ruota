  <div id="modalOverlay">

      <h2>Configura la Ruota</h2>

      <label for="optionsInput">Opzioni (separate da virgola):</label>

      <textarea id="optionsInput" rows="4">Pizza,Pasta,Gelato,Carne,Pesce,Insalata</textarea>

      <label for="fixedIndex">Risultato fisso (indice 0-n):</label>

  <script>

    // Variabili globali e setup

    const canvas = document.getElementById("wheelCanvas");

    const ctx = canvas.getContext("2d");

    let options = ["Pizza", "Pasta", "Gelato", "Carne", "Pesce", "Insalata"];

    let colors = ["#ff6666", "#66b3ff", "#99ff99", "#ffcc99", "#c266ff", "#ffff66"];

    const fixedIndexInput = document.getElementById("fixedIndex");

    let isSpinning = false;

    let angle = 0;

    // Disegna la ruota con spicchi dinamici

    function drawWheel() {

      const startCenter = 150;

      const radius = 150;

      const arc = (2 * Math.PI) / options.length;

      ctx.clearRect(0, 0, canvas.width, canvas.height);

      let startAngle = 0;

      for (let i = 0; i < options.length; i++) {

        ctx.beginPath();

        ctx.fillStyle = colors[i % colors.length];

        ctx.moveTo(startCenter, startCenter);

        ctx.arc(startCenter, startCenter, radius, startAngle, startAngle + arc);

        ctx.lineTo(startCenter, startCenter);

        ctx.fill();

        ctx.strokeStyle = "#fff";

        ctx.lineWidth = 2;

        ctx.stroke();

        // Disegna il testo centrato nello spicchio

        const textAngle = startAngle + arc / 2;

        ctx.save();

        ctx.translate(startCenter, startCenter);

        ctx.rotate(textAngle);

        ctx.textAlign = "center";

        ctx.textBaseline = "middle";

        ctx.font = `${Math.min(18, 120 / options.length)}px Comic Sans MS`;

        ctx.fillStyle = "#000";

        ctx.fillText(options[i], radius * 0.65, 0);

        ctx.restore();

        startAngle += arc;

      }

    }

    // Aggiorna la rotazione del canvas

    function updateCanvasRotation() {

      ctx.clearRect(0, 0, canvas.width, canvas.height);

      ctx.save();

      ctx.translate(150, 150);

      ctx.rotate(angle);

      ctx.translate(-150, -150);

      drawWheel();

      ctx.restore();

    }

    // Funzione per far girare la ruota (con controllo del risultato)

    function spinWheel() {

      if (isSpinning) return;

      isSpinning = true;

      const chosenIndex = parseInt(fixedIndexInput.value) % options.length;

      const degreesPerOption = 360 / options.length;

      // Calcola l'angolo in gradi al centro dello spicchio scelto

      const stopAngleDeg = 360 - (degreesPerOption * chosenIndex) - (degreesPerOption / 2);

      const extraRotation = 360 * 5; // 5 giri completi

      const randomOffset = Math.floor(Math.random() * 30) - 15;

      const finalAngleDeg = stopAngleDeg + extraRotation + randomOffset;

      let currentRotation = 0;

      const spinSpeed = 20;

      document.getElementById("spinSound").play();

      const spinInterval = setInterval(() => {

        currentRotation += spinSpeed;

        if (currentRotation >= finalAngleDeg) {

          currentRotation = finalAngleDeg;

          clearInterval(spinInterval);

          isSpinning = false;

          document.getElementById("winSound").play();

          highlightResult(chosenIndex);

        }

        angle = currentRotation * (Math.PI / 180);

        updateCanvasRotation();

      }, 30);

    }

    // Evidenzia lo spicchio vincente sulla ruota (senza scriverlo fuori)

    function highlightResult(index) {

      const arc = (2 * Math.PI) / options.length;

      const startAngle = arc * index;

      ctx.save();

      ctx.globalAlpha = 0.5;

      ctx.fillStyle = "red";

      ctx.beginPath();

      ctx.moveTo(150, 150);

      ctx.arc(150, 150, 150, startAngle, startAngle + arc);

      ctx.lineTo(150, 150);

      ctx.fill();

      ctx.restore();

    }

    // Aggiorna la ruota in base alle opzioni inserite nella modale

    function updateWheel() {

      const newOptions = document.getElementById("optionsInput").value

                           .split(",")

                           .map(o => o.trim())

                           .filter(o => o);

      if (newOptions.length < 2) {

        alert("Inserisci almeno due opzioni!");

        return;

      }

      options = newOptions;

      fixedIndexInput.max = options.length - 1;

      // Aggiorna i colori se necessario (ciclo i colori di base)

      drawWheel();

      toggleModal(); // chiude la modale dopo l'aggiornamento

    }

    // Incrementa il risultato predefinito (pulsante Reset)

    function resetWheel() {

      fixedIndexInput.value = (parseInt(fixedIndexInput.value) + 1) % options.length;

    }

    // Gestione della modale di configurazione

    function toggleModal() {

      const modal = document.getElementById("modalOverlay");

      modal.style.display = (modal.style.display === "flex") ? "none" : "flex";

    }

    // Disegna inizialmente la ruota

    drawWheel();

  </script>
